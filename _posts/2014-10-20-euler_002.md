Solution for [Problem 002] from [Project Euler]

{% highlight ruby linenos=table %}
seq = [1,2]
max = 4_000_000

while seq[-2] + seq[-1] < max
  seq << seq[-2] + seq[-1]
end

evens = seq.reject{ |n| n.odd? }
puts "Answer is: #{evens.inject(:+)}"
